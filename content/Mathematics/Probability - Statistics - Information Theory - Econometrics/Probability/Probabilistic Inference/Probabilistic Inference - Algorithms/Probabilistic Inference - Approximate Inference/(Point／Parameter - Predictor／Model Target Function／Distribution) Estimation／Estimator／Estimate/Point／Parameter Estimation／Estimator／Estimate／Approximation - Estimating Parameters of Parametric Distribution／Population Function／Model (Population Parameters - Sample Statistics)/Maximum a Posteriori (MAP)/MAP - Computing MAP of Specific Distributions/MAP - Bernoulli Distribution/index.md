- 𝑝<sub>𝑀𝐴𝑃</sub> = \[ ∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub>+ (𝛼-1)\] / \[ 𝑛 + 𝛽 + 𝛼 - 2) \]

where:
- 𝑝<sub>𝑀𝐴𝑃</sub> - estimated proportion of samples with value 1
- 𝑋<sub>𝑖</sub>- is a bernoulli variable taking 0 or 1 value
- 𝑛 - number of samples
- 𝛼 - can be thought as "number of prior samples having value 1"
- 𝛽 - can be thought as "number of prior samples having value 0"

### Computation
- 𝜃<sub>𝑀𝐴𝑃</sub> = 𝑎𝑟𝑔𝑚𝑎𝑥<sub>𝜃</sub>𝑙𝑜𝑔 ∏<sub>1≤𝑖≤𝑛</sub>\[𝐏(𝑋<sub>𝑖</sub>|𝜃)𝐏(𝜃)\]
- 𝜃<sub>𝑀𝐴𝑃</sub> = 𝑎𝑟𝑔𝑚𝑎𝑥<sub>𝜃</sub>∑<sub>1≤𝑖≤𝑛</sub>\[𝑙𝑜𝑔𝐏(𝑋<sub>𝑖</sub>|𝜃)\] + 𝑙𝑜𝑔𝐏(𝜃)

we have:
- <font style="color: rgb(0,128,0);">𝐏(𝑋<sub>𝑖</sub></font><font style="color: rgb(0,128,0);">|𝜃)</font> = [[Bernoulli Distribution|𝐵𝑒𝑟𝑛𝑜𝑢𝑙𝑙𝑖(𝑋𝑖|𝜃)]] = 𝜃<sup>𝑋<sub>𝑖</sub></sup>(1 - 𝜃)<sup>(1-𝑋<sub>𝑖</sub>)</sup>
- <font style="color: rgb(0,0,255);">𝐏(𝜃)</font> = [[Beta Distribution|𝐵𝑒𝑡𝑎(𝜃|𝛼,𝛽)]]= (1 - 𝜃)<sup>𝛽-1</sup>𝜃<sup>𝛼-1</sup>/ 𝐵(𝛼,𝛽)

thus:
- 𝜃<sub>𝑀𝐴𝑃</sub> = 𝑎𝑟𝑔𝑚𝑎𝑥<sub>𝜃</sub>\[∑<sub>1≤𝑖≤𝑛</sub>𝑙𝑜𝑔<font style="color: rgb(0,128,0);">𝐏(𝑋<sub>𝑖</sub>|𝜃)</font>\] + 𝑙𝑜𝑔<font style="color: rgb(0,0,255);">𝐏(𝜃)</font>
- 𝜃<sub>𝑀𝐴𝑃</sub> = 𝑎𝑟𝑔𝑚𝑎𝑥<sub>𝜃</sub>\[∑<sub>1≤𝑖≤𝑛</sub>𝑙𝑜𝑔<font style="color: rgb(0,128,0);">(</font><font style="color: rgb(0,128,0);">𝜃<sup>𝑋<sub>𝑖</sub></sup>(1 - 𝜃)<sup>(1-𝑋<sub>𝑖</sub>)</sup></font><font style="color: rgb(0,128,0);">)</font>\] + 𝑙𝑜𝑔<font style="color: rgb(0,0,255);">(</font><font style="color: rgb(0,0,255);">(1 - 𝜃)<sup>𝛽-1</sup>𝜃<sup>𝛼-1</sup>/ 𝐵(𝛼,𝛽)</font><font style="color: rgb(0,0,255);">)</font>
- <font style="color: rgb(0,0,255);"><font style="color: rgb(51,51,51);">𝜃<sub>𝑀𝐴𝑃</sub></font><font style="color: rgb(34,34,34);"> = </font><font style="color: rgb(34,34,34);">𝑎</font><font style="color: rgb(34,34,34);">𝑟</font><font style="color: rgb(34,34,34);">𝑔</font><font style="color: rgb(34,34,34);">𝑚</font><font style="color: rgb(34,34,34);">𝑎</font><font style="color: rgb(34,34,34);">𝑥</font><font style="color: rgb(51,51,51);"><sub>𝜃</sub></font><font style="color: rgb(34,34,34);">\[∑<sub>1≤𝑖≤𝑛</sub><font style="color: rgb(0,128,0);">𝑋</font><font style="color: rgb(0,128,0);"><sub>𝑖</sub>𝑙𝑜𝑔(</font><font style="color: rgb(0,128,0);">𝜃) + (1-𝑋<sub>𝑖</sub>)𝑙𝑜𝑔(1 - 𝜃)</font><font style="color: rgb(0,128,0);">)</font>\] <font style="color: rgb(51,51,51);">+</font> <font style="color: rgb(0,0,255);">(𝛽-1)𝑙𝑜𝑔(1 - 𝜃) + (𝛼-1)𝑙𝑜𝑔(𝜃) - 𝑙𝑜𝑔(𝐵(𝛼,𝛽))</font></font></font>

take derivative of \[∑<sub>1≤𝑖≤𝑛</sub><font style="color: rgb(0,128,0);">𝑋</font><font style="color: rgb(0,128,0);"><sub>𝑖</sub>𝑙𝑜𝑔(</font><font style="color: rgb(0,128,0);">𝜃) + (1-𝑋<sub>𝑖</sub>)𝑙𝑜𝑔(1 - 𝜃)</font><font style="color: rgb(0,128,0);">)</font>\] +<font style="color: rgb(0,0,255);">(𝛽-1)𝑙𝑜𝑔(1 - 𝜃) + (𝛼-1)𝑙𝑜𝑔(𝜃) - 𝑙𝑜𝑔(𝐵(𝛼,𝛽))</font>

derivative of first part
- 𝛿/𝛿<font style="color: rgb(0,0,255);"><font style="color: rgb(51,51,51);">𝜃</font> <font style="color: rgb(34,34,34);">\[∑</font><font style="color: rgb(51,51,51);"><sub>1≤𝑖≤𝑛</sub></font><font style="color: rgb(34,34,34);"><font style="color: rgb(0,128,0);">𝑋</font><font style="color: rgb(0,128,0);"><sub>𝑖</sub>𝑙𝑜𝑔(</font><font style="color: rgb(0,128,0);">𝜃) + (1-𝑋<sub>𝑖</sub>)𝑙𝑜𝑔(1 - 𝜃)</font><font style="color: rgb(0,128,0);">)</font></font><font style="color: rgb(34,34,34);">\]</font></font>
- <font style="color: rgb(0,128,0);">(1/𝜃)∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font><font style="color: rgb(0,128,0);">- (1/(1-𝜃))∑<sub>1≤𝑖≤𝑛</sub>1-𝑋<sub>𝑖</sub></font>

derivative of second part
- 𝛿/𝛿<font style="color: rgb(0,0,255);"><font style="color: rgb(51,51,51);">𝜃 <font style="color: rgb(0,0,255);">(𝛽-1)𝑙𝑜𝑔</font><font style="color: rgb(0,0,255);">(1 - 𝜃) + (𝛼-1)𝑙𝑜𝑔(𝜃) - 𝑙𝑜𝑔(𝐵(𝛼,</font><font style="color: rgb(0,0,255);">𝛽))</font></font></font>
- -<font style="color: rgb(0,0,255);">(𝛽 - 1)</font>/<font style="color: rgb(0,0,255);">(1 - 𝜃) + (𝛼-1)/(𝜃)</font>
- <font style="color: rgb(0,0,255);">(𝛼-1)/(𝜃) <font style="color: rgb(0,0,0);">- </font>(𝛽 - 1)<font style="color: rgb(0,0,0);">/</font>(1 - 𝜃)</font>

then set to 0 and solve for 𝜃:
- 0 = <font style="color: rgb(0,128,0);">(1/𝜃)</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font><font style="color: rgb(0,128,0);">- (1/(1-𝜃))∑<sub>1≤𝑖≤𝑛</sub>1-𝑋<sub>𝑖</sub></font>+ <font style="color: rgb(0,0,255);">(𝛼-1)/(𝜃) </font>- <font style="color: rgb(0,0,255);">(𝛽 - 1)</font>/<font style="color: rgb(0,0,255);">(1 - 𝜃)</font>
- 0 = <font style="color: rgb(255,0,0);">(1/𝜃)</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font><font style="color: rgb(0,128,0);">- <font style="color: rgb(255,0,0);">(1/(1-𝜃))</font>∑<sub>1≤𝑖≤𝑛</sub>1-𝑋<sub>𝑖</sub></font>+ <font style="color: rgb(0,0,255);">(𝛼-1)<font style="color: rgb(255,0,0);">(1/𝜃)</font> </font>- <font style="color: rgb(0,0,255);">(𝛽 - 1)<font style="color: rgb(255,0,0);">(1</font></font><font style="color: rgb(255,0,0);">/(1 - 𝜃))</font>
- 0 = <font style="color: rgb(255,0,0);">(1/𝜃)<font style="color: rgb(51,51,51);">\[<font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font> + <font style="color: rgb(0,0,255);">(𝛼-1)<font style="color: rgb(51,51,51);">\] - <font style="color: rgb(255,0,0);">(1/(1-𝜃))</font>\[<font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>1-𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖</font></sub>+ <font style="color: rgb(0,0,255);">(𝛽 - 1)</font>\]</font></font></font></font>
- <font style="color: rgb(255,0,0);"><font style="color: rgb(51,51,51);"><font style="color: rgb(255,0,0);">(1/(1-𝜃))</font>\[<font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>1-𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖</font></sub>+ <font style="color: rgb(0,0,255);">(𝛽 - 1)</font>\] = <font style="color: rgb(255,0,0);">(1/𝜃)</font>\[<font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font> + <font style="color: rgb(0,0,255);">(𝛼-1)<font style="color: rgb(51,51,51);">\]</font></font></font></font>
- <font style="color: rgb(255,0,0);"><font style="color: rgb(51,51,51);"><font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>1-𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖</font></sub>+ <font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,0,255);">(𝛽 - 1)</font> = <font style="color: rgb(255,0,0);">(1-𝜃)</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font> + <font style="color: rgb(255,0,0);">(1-𝜃)</font><font style="color: rgb(0,0,255);">(𝛼-1)</font></font></font>
- <font style="color: rgb(255,0,0);"><font style="color: rgb(51,51,51);"><font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>1-𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖</font></sub>+ <font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,0,255);">(𝛽 - 1)</font> = <font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖</font></sub><font style="color: rgb(255,0,0);"><font style="color: rgb(51,51,51);">-</font> 𝜃</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font> + <font style="color: rgb(0,0,255);">(𝛼-1)</font> <font style="color: rgb(255,0,0);"><font style="color: rgb(51,51,51);">-</font> 𝜃</font><font style="color: rgb(0,0,255);">(𝛼-1)</font></font></font>
- <font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>1-𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖</font></sub>+ <font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,0,255);">(𝛽 - 1) <font style="color: rgb(51,51,51);">+</font> <font style="color: rgb(255,0,0);">𝜃</font><font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋<sub>𝑖</sub></font> <font style="color: rgb(51,51,51);">+</font> <font style="color: rgb(255,0,0);">𝜃</font>(𝛼-1)<font style="color: rgb(51,51,51);"> = <font style="color: rgb(0,128,0);">∑<sub>1≤𝑖≤𝑛</sub>𝑋</font><sub><font style="color: rgb(0,128,0);">𝑖 </font></sub>+ <font style="color: rgb(0,0,255);">(𝛼-1)</font></font></font>

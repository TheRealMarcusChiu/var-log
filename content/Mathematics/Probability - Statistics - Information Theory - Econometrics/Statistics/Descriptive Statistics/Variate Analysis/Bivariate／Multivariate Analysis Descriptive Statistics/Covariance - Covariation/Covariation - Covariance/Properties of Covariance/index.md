```excerpt
- 𝐶𝑜𝑣(𝑋,𝑌) = 𝐄\[(𝑋-𝜇<sub>𝑋</sub>)(𝑌-𝜇<sub>𝑌</sub>)\] = 𝐄\[𝑋𝑌\] - 𝜇<sub>𝑋</sub>𝜇<sub>𝑌</sub>
- <strong>𝐶𝑜𝑣</strong>(𝑋,𝑌) = <strong>𝐶𝑜𝑣</strong>(𝑌,𝑋)
- <strong>𝐶𝑜𝑣</strong>(𝑎𝑋 + 𝑏, 𝑐𝑌 + 𝑑) = 𝑎𝑐<strong>𝐶𝑜𝑣</strong>(𝑋,𝑌)
- <strong>𝐶𝑜𝑣</strong>(𝑎𝑋 + 𝑏𝑌, 𝑐𝑍 + 𝑑𝑊) = 𝑎𝑐<strong>𝐶𝑜𝑣</strong>(𝑋,𝑍) + 𝑎𝑑<strong>𝐶𝑜𝑣</strong>(𝑋,𝑊) + 𝑏𝑐<strong>𝐶𝑜𝑣</strong>(𝑌,𝑍) + 𝑏𝑑<strong>𝐶𝑜𝑣</strong>(𝑌,𝑊)
- <strong>𝐶𝑜𝑣</strong>(𝑋,𝑌) = 0 <font style="color: rgb(128,128,128);">\# </font><font style="color: rgb(128,128,128);">for</font> [[Probability Independence (Marginal Independence - Conditional Independence)|independent]]<font style="color: rgb(128,128,128);"> 𝑋 and 𝑌</font>
```
